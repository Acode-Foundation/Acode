function sendMessage() { const input = document.getElementById('userInput'); const text = input.value.trim(); if (!text) return; const targetBotId = currentBot.id; const targetBotName = currentBot.name; // Lấy thêm tính cách của bot const targetBotPrompt = currentBot.prompt || 'Mặc định'; if (!chatHistories[targetBotId]) chatHistories[targetBotId] = []; chatHistories[targetBotId].push({ sender: 'user', text: text }); input.value = ''; renderMessages(); setTimeout(() => { // Đưa tính cách (prompt) vào nội dung phản hồi const replyText = `[${targetBotName} (${targetBotPrompt})]: Tớ đã nhận được tin nhắn "${text}".`; chatHistories[targetBotId].push({ sender: 'bot', text: replyText }); if (currentBot.id === targetBotId) { renderMessages(); } }, 800); }
